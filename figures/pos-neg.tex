  \inferrule[\defnamerule{pos-notin}{pos-$\notin$}]
    {\upsilon \notin \SV{t}}
    {\upsilon \pos t}
  \and\quad
  \inferrule[\defnamerule{neg-notin}{neg-$\notin$}]
    {\upsilon \notin \SV{t}}
    {\upsilon \neg t}
  \and\qquad
  \inferrule[\defnamerule{pos-prod}{pos-$\Pi$}]
    {\upsilon \notin \SV{t} \and \upsilon \pos u}
    {\upsilon \pos \prod{x}{t}{u}}
  \and\quad
  \inferrule[\defnamerule{neg-prod}{neg-$\Pi$}]
    {\upsilon \notin \SV{t} \and \upsilon \neg u}
    {\upsilon \neg \prod{x}{t}{u}}

  \inferrule[\defrule{pos-ind}]
    {\upsilon \notin \SV{\overline{a}} \and I ~ \text{inductive}}
    {\upsilon \pos \app{I^s}{\overline{a}}}

  \inferrule[\defrule{neg-coind}]
    {\upsilon \notin \SV{\overline{a}} \and I ~ \text{coinductive}}
    {\upsilon \neg \app{I^s}{\overline{a}}}
\end{mathpar}
